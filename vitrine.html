bibliotheque.php — synchronisation dossier ↔ BDD 
function syncFolderWithDB($dir, $type, $pdo) {
    $files = array_diff(scandir($dir), [".", ".."]);

    // Ajout des fichiers manquants en BDD
    foreach ($files as $file) {
        $check = $pdo->prepare("SELECT id FROM bibliotheque WHERE photos = :file");
        $check->execute(['file' => $file]);
        if (!$check->fetch()) {
            $stmt = $pdo->prepare("INSERT INTO bibliotheque (photos, fichier) VALUES (:file, :type)");
            $stmt->execute(['file' => $file, 'type' => $type]);
        }
    }
    // Suppression des fichiers qui n'existent plus
    // ...
}
scriptouioui.js — chargement lightbox + fetch description 
images.forEach(img => {
    img.addEventListener('click', (e) => {
        lightboxImg.src = img.src;
        lightbox.classList.remove('hidden');

        // Chargement dynamique de la fiche technique .txt
        const baseName = img.getAttribute('data-file');
        if (baseName) {
            fetch('fichier/' + baseName + '.txt')
                .then(r => r.text())
                .then(data => { lightboxDesc.innerHTML = formatDescription(data); });
        }
    });
});
